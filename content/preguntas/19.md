```py
a = "programando"
b = 'mucho'
c = '''un string
con múltiples
lineas'''

d = """Multiples con
     comillas dobles"""
e = ("Tres" " strings" " juntos")
f = "un string " + "concatenado"
g = ("Otra forma de string que nos permite "
    "utilizar más de una línea pero en verdad solo es una,"
    " lo que es muy útil para cumplir PEP-8 :)")

print(a)
print(b)
print(c)
print(d)
print(e)
print(f)
print(g)
```
